1 /**
  2  * @fileoverview script 异步载入模块.
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 define(['../event/event.js', '../bom/browser.js', '../type/lang.js'], function(event, browser, lang) {
  6 
  7 	var win = window,
  8 		doc = document;
  9 
 10 	/**
 11 	 * script 异步载入模块
 12 	 * @name script
 13 	 * @namespace
 14 	 */
 15 	return {
 16 
 17 		/**
 18 		 * @lends script
 19 		 * @static
 20 		 */
 21 
 22 		/**
 23 		 * 载入 script
 24 		 * @param { Object } cfg 配置项.
 25 		 * @return { HTMLelement } script node.
 26 		 */
 27 		get: function(cfg) {
 28 
 29 			var host = this,
 30 				node = host.create(cfg.url);
 31 
 32 			event.add(node, browser.isIE < 9 ? 'readystatechange' : 'load', cfg.success || function() {});
 33 
 34 			return node;
 35 
 36 		},
 37 
 38 		/**
 39 		 * 创建 script 标签
 40 		 * @param { String } url script 地址.
 41 		 * @return { HTMLelement } script node.
 42 		 */
 43 		create: function(url) {
 44 
 45 			var s = doc.createElement('script');
 46 
 47 			s.src = url;
 48 
 49 			s.charset = 'utf-8';
 50 
 51 			return doc.head.appendChild(s);
 52 
 53 		},
 54 
 55 		/**
 56 		 * 移除 script 标签
 57 		 * @param { HTMLelement } node script node.
 58 		 */
 59 		remove: function(node) {
 60 
 61 			node.parentNode.removeChild(node);
 62 
 63 		},
 64 
 65 		/**
 66 		 * jsonp
 67 		 * @param { Object } cfg 配置项.
 68 		 */
 69 		jsonp: function(cfg) {
 70 
 71 			var host = this,
 72 				s,
 73 				url = cfg.url,
 74 				success = cfg.success,
 75 				isFunction = lang.isFunction(success),
 76 				id;
 77 
 78 			if (isFunction) {
 79 
 80 				id = 'nature_jsonp_' + new Date().getTime();
 81 				win[id] = success;
 82 
 83 			} else {
 84 
 85 				id = success;
 86 
 87 			}
 88 
 89 			s = host.get({
 90 
 91 				url: url + (url.indexOf('?') > -1 ? '&' : '?') + 'callback=' + id,
 92 
 93 				success: function() {
 94 
 95 					host.remove(s);
 96 
 97 					isFunction && delete win[id];
 98 
 99 				}
100 
101 			});
102 
103 		}
104 
105 	};
106 
107 });
