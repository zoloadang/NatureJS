 17 	/**
 18 	 * 扩展类的方法
 19 	 * @param { Function } r 需要接收扩展的 constructor.
 20 	 * @param { Function } s 需要提供扩展的 constructor.
 21 	 * @param { Object } px 需要添加的原型方法集合.
 22 	 * @param { Object } sx 需要添加的静态方法集合.
 23 	 * @return { Function } 构造器 r.
 24 	 * @spec extend a constructor
 25 	 * @example
 26 	 *      var A = function() {};
 27 	 *      var B = function() {};
 28 	 *      extend(B, A);
 29 	 *      var c = new B;
 30 	 *      c instanceof A => true
 31 	 *      c instanceof B => true
 32 	 */
 33 	function extend(r, s, px, sx) {
 34 
 35 		if (!r) {
 36 			return false;
 37 		}
 38 
 39 		var sp,
 40 			proto = {};
 41 
 42 		if (s) {
 43 
 44 			sp = s.prototype;
 45 			proto = object.create(sp);
 46 			//处理来源函数
 47 			if (Object != s && s != sp.constructor) {
 48 				sp.constructor = s;
 49 			}
 50 			r.superclass = sp;
 51 
 52 		}
 53 
 54 		r.prototype = proto;
 55 		proto.constructor = r;
 56 
 57 		if (px) {
 58 			object.mixin(proto, px);
 59 		}
 60 
 61 		if (sx) {
 62 			object.mixin(r, sx);
 63 		}
 64 
 65 		return r;
 66 
 67 	}
 68 
 69 	return extend;
 70
