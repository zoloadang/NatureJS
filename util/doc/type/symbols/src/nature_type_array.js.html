 24 		/**
 25 		 * 正向查找元素在数组中的第一个位置
 26 		 * @param { Array } arr 数组.
 27 		 * @param { Object } value 元素.
 28 		 * @param { Number } fromindex 开始位置.
 29 		 * @param { Boolean } fromlast 是否从后开始查找.
 30 		 * @return { Number } 索引.
 31 		 * @see https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Array/indexOf
 32 		 * @spec locates the first index of the provided value in the passed array.
 33 		 * @example
 34 		 *      array.indexOf([1, 2, 1, 4], 1) => 0
 35 		 *      array.indexOf([1, 2, 1, 4], 1, 1) => 2
 36 		 *      array.indexOf([1, 2, 1, 4], 1,  -1) => -1
 37 		 *      array.indexOf([1, 2, 1, 4], 1, -2) => 2
 38 		 */
 39 		indexOf: function(arr, value, fromindex, fromlast) {
 40 
 41 			if (proto.indexOf) {
 42 
 43 				return arr.indexOf(value, lang.isUndefined(fromindex) ? 0 : fromindex);
 44 
 45 			}
 46 
 47 			var end = arr.length,
 48 				step,
 49 				i = !lang.isUndefined(fromindex) ? (fromindex < 0 ? (fromindex + end) : fromindex) : -1;
 50 
 51 			if (!fromlast) {
 52 
 53 				step = 1;
 54 				i = i == -1 ? 0 : i;
 55 
 56 			} else {
 57 
 58 				i = i == -1 ? (end - 1) : i;
 59 				end = step = -1;
 60 
 61 			}
 62 
 63 			for (; i != end; i += step) {
 64 
 65 				if (arr[i] == value) {
 66 
 67 					return i;
 68 
 69 				}
 70 
 71 			}
 72 
 73 			return -1;
 74 
 75 		},

 84 		 * @spec locates the last index of the provided value in the passed array.
 85 		 * @example

 87 		 *      array.lastIndexOf([1, 2, 1, 4], 1, 1) => 0

 89 		 *      array.lastIndexOf([1, 2, 1, 4], 1, -2) => 2

 91 		lastIndexOf: function(arr, value, fromindex) {
 92

 94 
 95 				return arr.lastIndexOf(value, lang.isUndefined(fromindex) ? (arr.length - 1) : fromindex);
 96 
 97 			}
 98 
 99 			return this.indexOf(arr, value, fromindex, true);

103 		/**
104 		 * 循环数组, 并对数组元素进行函数调用
105 		 * @param { Array } arr 数组.

107 		 * @param { Object } thisObject 作用域.
108 		 * @see https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/array/foreach
109 		 * @spec for every item in arr, callback is invoked.
110 		 * @example
111 		 *      var str = '';
112 		 *      array.forEach([1, 2], function(item, i, a) {
113 		 *      	str += item + ',' + i + ',' + a.join(',');	
114 		 *      });	
115 		 *      str => '1,0,1,22,1,1,2'
116 		 */
117 		forEach: function(arr, callback, thisObject) {
118 
119 			if (proto.forEach) {
120 
121 				return arr.forEach(callback, thisObject);
122 
123 			}
124 
125 			for (var i = 0, len = arr.length; i < len; ++i) {
126 
127 				callback.call(thisObject || win, arr[i], i, arr);
128 
129 			}

133 		/**
134 		 * 是否所有数组元素都能通过函数校验
135 		 * @param { Array } arr 数组.
136 		 * @param { Function } callback 校验函数.
137 		 * @param { Object } thisObject 作用域.
138 		 * @return { Boolean } 如果全部通过校验则返回 true, 否则返回 false.
139 		 * @see https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Objects/Array/every
140 		 * @spec Determines whether or not every item in arr satisfies the condition implemented by callback.
141 		 * @example
142 		 *      var a = array.every([1, 2], function(i) {
143 		 *      	 		return 'number' == typeof i;
144 		 *      	 	}),	
145 		 *       	b = array.every([1, '2'], function(i) {
146 		 *       		    return 'number' == typeof i;
147 		 *       	    });
148 		 *      a => true
149 		 *      b => false
150 		 */
151 		every: function(arr, callback, thisObject) {
152 
153 			if (proto.every) {
154 
155 				return arr.every(callback, thisObject);
156 
157 			}
158 
159 			return this._everyOrSome(true, arr, callback, thisObject);
160

163 		/**
164 		 * 是否所有数组元素都能通过函数校验
165 		 * @private
166 		 * @param { Boolean } every 是否检验所有.
167 		 * @param { Array } arr 数组.
168 		 * @param { Function } callback 校验函数.
169 		 * @param { Object } thisObject 作用域.
170 		 * @return { Boolean }
171 		 */
172 		_everyOrSome: function(every, arr, callback, thisObject) {
173 
174 			var i = 0,
175 				len = arr.length,
176 				result;
177 
178 			for (; i < len; ++i) {
179 
180 				result = !!callback.call(thisObject || win, arr[i], i, arr);
181 
182 				if (every ^ result) {
183 
184 					return result; // Boolean
185 
186 				}
187 
188 			}
189 
190 			return every; // Boolean
191 
192 		},

194 		/**
195 		 * 是否有数组元素能够通过函数校验
196 		 * @param { Array } arr 数组.
197 		 * @param { Function } callback 校验函数.
198 		 * @param { Object } thisObject 作用域.
199 		 * @return { Boolean } 如果有元素通过校验则返回 true, 否则返回 false.
200 		 * @see https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Objects/Array/some
201 		 * @spec Determines whether or not any item in arr satisfies the condition implemented by callback.
202 		 * @example
203 		 *      var a = array.some([1, 2], function(i) {
204 		 *      			return 'number' == typeof i;
205 		 *      		}),
206 		 *      	b = array.some([1, '2'], function(i) {
207 		 *      			return 'number' == typeof i;
208 		 *      		}),	
209 		 *      	c = array.some(['1', '2'], function(i) {
210 		 *      			return 'number' == typeof i;
211 		 *      		});
212 		 *      a => true
213 		 *      b => true
214 		 *      c => false
215 		 */
216 		some: function(arr, callback, thisObject) {
217 
218 			if (proto.some) {
219 
220 				return arr.some(callback, thisObject);
221 
222 			}
223 
224 			return this._everyOrSome(false, arr, callback, thisObject);

228 		/**
229 		 * 对数组每个元素作为函数参数运行, 返回由结果组成的数组
230 		 * @param { Array } arr 数组.
231 		 * @param { Function } callback 函数.
232 		 * @param { thisObject } 作用域.
233 		 * @return { Array } 给定函数运行结果的集合.
234 		 * @see https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Objects/Array/map
235 		 * @spec applies callback to each element of arr and returns an Array with the results.
236 		 * @example
237 		 *      var r = array.map([1, 2, 3], function(i) {
238 		 *      	        return i * 2;	
239 		 *      		});
240 		 *      r.join(',') => '2,4,6'
241 		 */
242 		map: function(arr, callback, thisObject) {
243 
244 			if (proto.map) {
245 				return arr.map(callback, thisObject);
246 			}
247 
248 			var i = 0,
249 				len = arr.length,
250 				result = [];
251 
252 			for (; i < len; ++i) {
253 
254 				result.push(callback.call(thisObject || win, arr[i], i, arr));
255 
256 			}
257 
258 			return result;
259

264 		 * @param { Array } arr 数组.
265 		 * @param { Function } callback 函数.
266 		 * @param { thisObject } 作用域.
267 		 * @return { Array } 通过校验的元素数组.
268 		 * @see https://developer.mozilla.org/en/Core_JavaScript_1.5_Reference/Objects/Array/filter
269 		 * @spec Returns a new Array with those items from arr that match the condition implemented by callback.
270 		 * @example
271 		 *      var r = array.filter([1, 2, '3'], function(i) {
272 	 	 *      			return 'number' == typeof i;
273 		 *      		});
274 		 *      r.join(',') => '1,2'

276 		filter: function(arr, callback, thisObject) {
277 
278 			if (proto.filter) {
279 				return arr.filter(callback, thisObject);
280 			}
281 
282 			var i = 0,
283 				len = arr.length,
284 				result = [];
285 
286 			for (; i < len; ++i) {
287 
288 				if (!!callback.call(thisObject || win, arr[i], i, arr)) {
289 
290 					result.push(arr[i]);
291 
292 				}
293 
294 			}
295 
296 			return result;

298 		}
