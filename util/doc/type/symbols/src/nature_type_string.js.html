1 /**
  2  * @fileoverview string 模块, 常用 String 方法.

  5 
  6 define(function() {
  7 
  8 	/**
  9 	 * string 模块
 10 	 * @name string
 11 	 * @namespace
 12 	 */
 13 
 14 	return {
 15 
 16 		/**
 17 		 * @lends string
 18 		 * @static
 19 		 */
 20 
 21 		/**
 22 		 * 去除字符串两边空格
 23 		 * @memberOf string
 24 		 * @param { String } str 需要清理的字符串.
 25 		 * @param { Boolean } isBigText 是否是大字符串.
 26 		 * @return { String } 返回清除后的字符串.
 27 		 * @spec romove both ends of the blank string.
 28 		 * @example
 29 		 *      string.trim(' test ') => 'test';
 30 		 *      string.trim('      I am nanzhi.') => 'I am nanzhi.';
 31 		 *      string.trim('      I am nanzhi.     ') => 'I am nanzhi.';
 32 		 *      string.trim('      I am nanzhi.     ', true) => 'I am nanzhi.';
 33 		 */
 34 		trim: function(str, isBigText) {
 35 
 36 			if (String.prototype.trim) {
 37 				return str.trim();
 38 			}
 39 
 40 			if (!isBigText) {
 41 				return str.replace(/^\s\s*/, '').replace(/\s\s*$/, '');
 42 			}
 43 
 44 			str = str.replace(/^\s+/, '');
 45 
 46 			for (var i = str.length - 1; i >= 0; i--) {
 47 				if (/\S/.test(str.charAt(i))) {
 48 					str = str.substring(0, i + 1);
 49 					break;
 50 				}
 51 			}
 52 
 53 			return str;
 54 
 55 		},
 56 
 57 		/**
 58 		 * 模板/对象匹配替换
 59 		 * @param { String } template 需要替换的模板.
 60 		 * @param { Object } data 相匹配的数据对象.
 61 		 * @param { String } prefix 匹配的前缀, 可选.
 62 		 * @param { String } postfix 匹配的后缀, 可选.
 63 		 * @param { String } postfix 匹配的后缀.
 64 		 * @return { String } 替换后的字符串.
 65 		 * @spec replace specific character by data.
 66 		 * @example
 67 		 *      var str = '{person} is a {role}.',
 68 		 *      	data = { person: 'Nanzhi', role: 'Developer' };
 69 		 *      string.substitute(str, data) => 'Nanzhi is a Developer.';
 70 		 */
 71 		substitute: function(template, data, prefix, postfix) {
 72 
 73 			var pre = prefix || '{',
 74 				post = postfix || '}',
 75 				reg = new RegExp(pre + '(.+?)' + post, 'g');
 76 
 77 			return template.replace(reg, function(match, value) {
 78 
 79 						return data[value];
 80 
 81 				   });
 82 
 83 		},
 84 
 85 		/**
 86 		 * 将首字母大写
 87 		 * @param { String } str 字符串.
 88 		 * @return { String } 转换后的字符串.
 89 		 */
 90 		toUpper: function(str) {
 91 
 92 			return str.substr(0, 1).toUpperCase() + str.substr(1);
 93 
 94 		}
 95 
 96 
 97 	};
 98 
 99 });
