 27 		 * @return { Boolean } 如果是字符串, 返回 true, 否则返回 false.
 28 		 * @spec check it is or not a string.
 29 		 * @example
 30 		 *          lang.isString('nanzhi') => true;

 32 		 *      lang.isString(12) => false;

 54 		 *      lang.isNumber(Array) => false;

 56 		isNumber: function(it) {

 83 		/**
 84 		 * 是否是类似数组的东西
 85 		 * @param { Anything } it
 86 		 * @return { Boolean } 它比 isArray 更宽松一点, 如果是类似数组的东西(节点, arguments)或者数组, 返回 true, 否则返回 false.
 87 		 * @spec check it is or not like array.
 88 		 * @example
 89 		 *      lang.isArrayLike([1, 2]) => true;
 90 		 *      lang.isArrayLike(arguments) => true;
 91 		 *      lang.isArrayLike(document.getElementsByTagName('head')) => true;

 98 		isArrayLike: function(it) {
 99 
100 			var host = this;
101 
102 			return it && !host.isString(it) && !host.isNumber(it) && !host.isFunction(it) &&
103 			       !(it.tagName && 'form' === it.tagName.toLowCase()) &&
104 				   (host.isArray(it) || isFinite(it.length));//@FIXME if length is a custom property, e.x: -1 or 1.2

112 		 * @spec check it is or not a object.

114 		 *      lang.isObject(null) => true;
115 		 *      lang.isObject([1, 2]) => true;
116 		 *      lang.isObject(Array) => true;
117 		 *      lang.isObject(function() {}) => true;
118 		 *      lang.isObject(arguments) => true;
119 		 *      lang.isObject('nanzhi') => false;
120 		 *      lang.isObject(undefined) => false;
121 		 *      lang.isObject(12) => false;
122 		 */
123 		isObject: function(it) {
124

126 
127 			return !host.isUndefined(it) && (host.isArray(it) || host.isFunction(it) || host.isNull(it) || 'object' === typeof it);
128 
129 		},
130 
131 		/**
132 		 * 是否是函数
133 		 * @param { Anything } it
134 		 * @return { Boolean } 如果是函数, 返回 true, 否则返回 false.
135 		 * @spec check it is or not a Function.
136 		 * @example
137 		 *      lang.isFunction(function() {}) => true;
138 		 *      lang.isFunction('nanzhi') => false;
139 		 *      !!lang.isFunction(undefined) => false;
140 		 *      lang.isFunction(12) => false;
141 		 *      !!lang.isFunction(null) => false;
142 		 *      lang.isFunction([1, 2]) => false;

144 		 *      lang.isFunction(arguments) => false;

154 		 * @param { Anything } it
155 		 * @return { Boolean } 如果是 null, 返回 true, 否则返回 false.
156 		 * @spec heck it is or not null.
157 		 * @example
158 		 *      lang.isNull(null) => true;
159 		 *      lang.isNull('nanzhi') => false;
160 		 *      lang.isNull(undefined) => false;
161 		 *      lang.isNull(12) => false;
162 		 *      lang.isNull([1, 2]) => false;
163 		 *      lang.isNull(Array) => false;
164 		 *      lang.isNull(arguments) => false;
165 		 */
166 		isNull: function(it) {
167

169 
170 		},
171 
172 		/**
173 		 * 是否是 undefined
174 		 * @param { Anything } it
175 		 * @return { Boolean } 如果是 undefined, 返回 true, 否则返回 false.
176 		 * @spec heck it is or not undefined.
177 		 * @example
178 		 *      lang.isUndefined(undefined) => true;
179 		 *      lang.isUndefined('nanzhi') => false;
180 		 *      lang.isUndefined(12) => false;
181 		 *      lang.isUndefined(null) => false;
182 		 *      lang.isUndefined([1, 2]) => false;
183 		 *      lang.isUndefined(Array) => false;
184 		 *      lang.isUndefined(arguments) => false;

194 		 * @param { ArrayLike } obj arrayLink.

196 		 * @return { Array }
197 		 */

202 		},
203 
204 		/**
205 		 * 绑定事件作用域以及参数
206 		 * @param { Object } scope 作用域.
207 		 * @param { Function | String } method 函数或者对象方法.
208 		 * @param { ... } 函数参数.
209 		 * @return { Function } 绑定函数作用域和参数的函数.
210 		 */
211 		hitch: function(scope, method/*, ... */) {
212 
213 			var host = this,
214 				args = arguments,
215 				pre = [],
216 				context = scope || win,
217 				fp = host.isString(method) ? scope[method] : method;
218 
219 			if (args.length > 2) {
220 
221 				pre = host.toArray(args, 2);
222 
223 			}
224 
225 			return function() {
226 
227 				var args = host.toArray(arguments);
228 
229 				return fp && fp.apply(context, pre.concat(args));
230 
231 			}
232 
233 		}
