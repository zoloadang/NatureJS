 26 		 * @param { Anything } it
 27 		 * @return { Boolean } 如果是字符串, 返回 true, 否则返回 false.
 28 		 */
 29 		isString: function(it) {
 30 
 31 			return it && ('string' === typeof it || it instanceof String);
 32 
 33 		},
 34 
 35 		/**
 36 		 * 是否是数字
 37 		 * @param { Anything } it
 38 		 * @return { Boolean } 如果是数字, 返回 true, 否则返回 false.
 39 		 */
 40 		isNumber: function(it) {

 43 
 44 		},
 45 
 46 		/**
 47 		 * 是否是数组
 48 		 * @param { Anything } it
 49 		 * @return { Boolean } 如果是数组, 返回 true, 否则返回 false.
 50 		 */
 51 		isArray: function(it) {
 52 
 53 			return it && ('array' === typeof it || it instanceof Array);
 54 
 55 		},
 56 
 57 		/**
 58 		 * 是否是类似数组的东西
 59 		 * @param { Anything } it
 60 		 * @return { Boolean } 它比 isArray 更宽松一点, 如果是类似数组的东西(节点, arguments)或者数组, 返回 true, 否则返回 false.
 61 		 */
 62 		isArrayLike: function(it) {
 63 
 64 			var host = this;
 65 
 66 			return it && !host.isString(it) && !host.isNumber(it) && !host.isFunction(it) &&
 67 			       !(it.tagName && 'form' === it.tagName.toLowCase()) &&
 68 				   (host.isArray(it) || isFinite(it.length));//@FIXME if length is a custom property, e.x: -1 or 1.2
 69 
 70 		},
 71 
 72 		/**
 73 		 * 是否是对象
 74 		 * @param { Anything } it
 75 		 * @return { Boolean } 如果是对象, 返回 true, 否则返回 false.
 76 		 */
 77 		isObject: function(it) {
 78 
 79 			var host = this;
 80 
 81 			return !host.isUndefined(it) && (host.isArray(it) || host.isFunction(it) || host.isNull(it) || 'object' === typeof it);
 82 
 83 		},
 84 
 85 		/**
 86 		 * 是否是函数
 87 		 * @param { Anything } it
 88 		 * @return { Boolean } 如果是函数, 返回 true, 否则返回 false.
 89 		 */
 90 		isFunction: function(it) {
 91 
 92 			return it && '[object Function]' === opts.call(it);
 93 
 94 		},
 95 
 96 		/**
 97 		 * 是否是空对象
 98 		 * @param { Anything } it
 99 		 * @return { Boolean } 如果是 null, 返回 true, 否则返回 false.
100 		 */
101 		isNull: function(it) {
102 
103 			return !this.isUndefined(it) && null === it;
104 
105 		},
106 
107 		/**
108 		 * 是否是 undefined
109 		 * @param { Anything } it
110 		 * @return { Boolean } 如果是 undefined, 返回 true, 否则返回 false.
111 		 */
112 		isUndefined: function(it) {
113 
114 			return 'undefined' === typeof it;
115 
116 		},
117 
118 		/**
119 		 * 转换 arrayLike 为 array
120 		 * @param { ArrayLike } obj arrayLink.
121 		 * @param { Number } fromIndex 开始位置.
122 		 * @return { Array }
123 		 */
124 		toArray: function(obj, fromIndex) {
125 
126 			return Array.prototype.slice.call(obj, fromIndex || 0);
127 
128 		},
129 
130 		/**
131 		 * 绑定事件作用域以及参数
132 		 * @param { Object } scope 作用域.
133 		 * @param { Function | String } method 函数或者对象方法.
134 		 * @param { ... } 函数参数.
135 		 * @return { Function } 绑定函数作用域和参数的函数.
136 		 */
137 		hitch: function(scope, method/*, ... */) {
138 
139 			var host = this,
140 				args = arguments,
141 				pre = [],
142 				context = scope || win,
143 				fp = host.isString(method) ? scope[method] : method;
144 
145 			if (args.length > 2) {
146 
147 				pre = host.toArray(args, 2);
148 
149 			}
150 
151 			return function() {
152 
153 				var args = host.toArray(arguments);
154 
155 				return fp && fp.apply(context, pre.concat(args));
156 
157 			}
158 
159 		}
160 
161 	};
162 
163 });
