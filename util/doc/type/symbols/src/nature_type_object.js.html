 24 		/**
 25 		 * 合并对象
 26 		 * @private
 27 		 * @param { Object } target 目标对象.
 28 		 * @param { Object } source 源对象.
 29 		 * @return { Object } target.
 30 		 */
 31 		_mixin: function(target, source) {
 32 
 33 			if (source) {
 34 
 35 				var i,
 36 					_s,
 37 					name,
 38 					empty = {},
 39 					len = extraNames.length;
 40 
 41 				for (name in source) {
 42 
 43 					_s = source[name];
 44 
 45 					if ( !target[name] || target[name] != _s && (!(name in empty) || _s != empty[name])) {
 46 
 47 						target[name] = source[name];
 48 
 49 					}
 50 
 51 				}
 52 
 53 				//针对 ie 浏览器
 54 				for (i = 0; i < len; i++) {
 55 
 56 					name = extraNames[i];
 57 					_s = source[name];
 58 
 59 					if ( !target[name] || target[name] != _s && (!(name in empty) || _s != empty[name])) {
 60 						target[name] = _s;
 61 					}
 62 
 63 				}
 64 
 65 			}
 66 
 67 			return target;
 68 
 69 		},

 71 		/**

 73 		 * @param { Object } obj 目标对象.

 75 		 * @return { Object } obj.
 76 		 * @spec Mix many object to one object.
 77 		 * @example
 78 		 *      var t = { a: 'b' },
 79 		 *      	a = { c: 'd' },
 80 		 *      	b = { e: 'f' };
 81 		 *      object.mixin(t, a, b)
 82 		 *      t['c'] => 'd';
 83 		 *      t['e'] => 'f';
 84 		 */
 85 		mixin: function(obj, props) {
 86 
 87 			var host = this,
 88 				args = arguments,
 89 				i = 1,
 90 				o = obj || {},
 91 				len = args.length;
 92 
 93 			for (; i < len; i++) {
 94 
 95 				host._mixin(o, args[i]);
 96 
 97 			}
 98 
 99 			return o;
100

125 		/**
126 		 * 转换对象为字符串
127 		 * @param { Object } obj 对象.
128 		 * @param { Array } symbol 分隔符, 数组中的第一个为 key, value 之间的间隔符, 第二个为各值之间的间隔.
129 		 * @param { Boolean } needEncode 是否需要编码中文.
130 		 * @return { String } 转换成的字符串.
131 		 * @spec Translate an object to string.
132 		 * @example
133 		 *      object.toString({ a: 'b', c: 'd' }) => 'a:b,c:d'
134 		 *      object.toString({ a: 'b', c: 'd' }, ['=', '&']) => 'a=b&c=d'
135 		 */
136 		toString: function(obj, symbol, needEncode) {
137 
138 			var i,
139 				arr = [],
140 				sym = symbol || [':', ','];
141 
142 			for (i in obj) {
143 
144 				arr.push(i + sym[0] + (!needEncode ? obj[i] : escape(obj[i].toString())));
145 
146 			}
147 
148 			return arr.join(sym[1]);
149 
150 		},

152 		/**
153 		 * 转换字符串到对象
154 		 * @param { String } str 字符串.
155 		 * @param { Array } symbol 分隔符.
156 		 * @param { Boolean } needDecode 是否需要解码.
157 		 * @return { Object } 对象.
158 		 * @spec Translate a string to object.
159 		 * @example
160 		 *      var o1 = object.toObject('a:b,c:d'),
161 		 *      	o2 = object.toObject('a=b&c=d', ['=', '&']);
162 		 *      o1['a'] => 'b';
163 		 *      o1['c'] => 'd';
164 		 *      o2['a'] => 'b';
165 		 *      o2['c'] => 'd';
166 		 */
167 		toObject: function(str, symbol, needDecode) {
168 
169 			var	obj = {},
170 				sym = symbol || [':', ','],
171 				arr = str.split(sym[1]),
172 				i = 0,
173 				len = arr.length;
174 
175 			for (; i < len; i++) {
176 
177 				var kv = arr[i].split(sym[0]);
178 				obj[kv[0]] = !needDecode ? kv[1] : unescape(kv[1]);
179 
180 			}
181 
182 			return obj;
183 
184 		},

186 		/**

188 		 * @private
189 		 * @param { String } parts 名字.

197 				len = parts.length,
198 				i = 0;
199 
200 			for (; i < len; i++) {
201 
202 				if (obj[parts[i]]) {
203 
204 					obj = obj[parts[i]];
205 					continue;
206 
207 				} else {
208 
209 					obj = create ? (obj[parts[i]] = {}) : 'undefined';
210 
211 				}
212 
213 			}
214 
215 			return obj;

275 
276 			}

278 			var keys = [],
279 				p;
280 
281 			for (p in obj) {
282 
283 				if (obj.hasOwnProperty(p)) {

286 
287 				}
288 
289 			}

291 			return keys;
