  6 define(function() {
  7 
  8 	/**
  9 	 * object 模块
 10 	 * @name object
 11 	 * @namespace
 12 	 */
 13 	return {
 14 
 15 		/**
 16 		 * @lends object
 17 		 * @static
 18 		 */
 19 
 20 		/**
 21 		 * 转换对象为字符串
 22 		 * @param { Object } obj 对象.
 23 		 * @param { Array } symbol 分隔符, 数组中的第一个为 key, value 之间的间隔符, 第二个为各值之间的间隔.
 24 		 * @param { Boolean } needEncode 是否需要编码中文.
 25 		 * @return { String } 转换成的字符串.
 26 		 * @example
 27 		 * 		object.toString({ a: 'b', c: 'd' }) //a:b,c:d
 28 		 * 		object.toString({ a: 'b', c: 'd' }, ['=', '&']) //a=b&c=d
 29 		 */
 30 		toString: function(obj, symbol, needEncode) {
 31 
 32 			var i,
 33 				arr = [],
 34 				sym = symbol || [':', ','];
 35 
 36 			for (i in obj) {
 37 
 38 				arr.push(i + sym[0] + (!needEncode ? obj[i] : escape(obj[i])));
 39 
 40 			}
 41 
 42 			return arr.join(sym[1]);
 43 
 44 		},
 45 
 46 		/**
 47 		 * 转换字符串到对象
 48 		 * @param { String } str 字符串.
 49 		 * @param { Array } symbol 分隔符.
 50 		 * @param { Boolean } needDecode 是否需要解码.
 51 		 * @return { Object } 对象.
 52 		 * @example
 53 		 * 		object.toObject('a:b,c:d') //{a:'b',c:'d'}
 54 		 * 		object.toObject('a=b&c=b', ['=', '&']) //{a:'b',c:'d'}
 55 		 */
 56 		toObject: function(str, symbol, needDecode) {
 57 
 58 			var	obj = {},
 59 				sym = symbol || [':', ','],
 60 				arr = str.split(sym[0]),
 61 				i = 0,
 62 				len = arr.length;
 63 
 64 			for (; i < len; i++) {
 65 
 66 				var kv = arr[i].split(sym[1]);
 67 				obj[kv[0]] = !needDecode ? kv[1] : unescape(kv[1]);
 68 
 69 			}
 70 
 71 			return obj;
 72 
 73 		},
 74 
 75 		/**
 76 		 * 转换对象为 url 参数
 77 		 * @param { Object } obj 对象.
 78 		 * @return { String } url 参数字符串.
 79 		 */
 80 		param: function(obj) {
 81 
 82 			return this.toString(obj, ['=', '&'], true);
 83 
 84 		},
 85 
 86 		/**
 87 		 * 转换 url 参数为字符串
 88 		 * @param { String } str url 参数字符串.
 89 		 * @return { Object } 对象.
 90 		 */
 91 		unparam: function(str) {
 92 
 93 			return this.toObject(str, ['=', '&'], true);
 94 
 95 		}
 96 
 97 
 98 	};
 99 
100 });
