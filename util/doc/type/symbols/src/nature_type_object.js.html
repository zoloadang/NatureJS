  6 define(['../type/lang.js'], function(lang) {
  7 
  8 	//对象自有方法, 属性
  9 	var win = window,
 10 		extraNames = ['hasOwnProperty', 'valueOf', 'isPrototypeOf', 'propertyIsEnumerable', 'toLocaleString', 'toString', 'constructor'];
 11 
 12 	/**
 13 	 * object 模块
 14 	 * @name object
 15 	 * @namespace
 16 	 */
 17 	return {

 19 		/**
 20 		 * @lends object
 21 		 * @static

 23 
 24 		/**
 25 		 * 合并对象
 26 		 * @private
 27 		 * @param { Object } target 目标对象.
 28 		 * @param { Object } source 源对象.
 29 		 * @return { Object } target.
 30 		 */
 31 		_mixin: function(target, source) {
 32 
 33 			if (source) {
 34 
 35 				var i,
 36 					_s,
 37 					name,
 38 					empty = {},
 39 					len = extraNames.length;
 40 
 41 				for (name in source) {
 42 
 43 					_s = source[name];
 44 
 45 					if (
 46 						!target[name] ||
 47 						target[name] != _s && (!(name in empty) || _s != empty[name])
 48 					) {
 49 
 50 						target[name] = source[name];
 51 
 52 					}
 53 
 54 				}
 55 
 56 				//针对 ie 浏览器
 57 				for (i = 0; i < len; i++) {
 58 
 59 					name = extraNames[i];
 60 					_s = source[name];
 61 
 62 					if (
 63 						!target[name] ||
 64 						target[name] != _s && (!(name in empty) || _s != empty[name])
 65 					) {
 66 						target[name] = _s;
 67 					}
 68 
 69 				}
 70 
 71 			}
 72 
 73 			return target;
 74 
 75 		},
 76 
 77 		/**
 78 		 * 合并对象元素到目标对象
 79 		 * @param { Object } obj 目标对象.
 80 		 * @param { Object } props 源对象.
 81 		 * @return { Object } obj.
 82 		 * @spec Mix many object to one object.
 83 		 * @example
 84 		 *      var t = { a: 'b' },
 85 		 *      	a = { c: 'd' },
 86 		 *      	b = { e: 'f' };
 87 		 *      object.mixin(t, a, b)
 88 		 *      t['c'] => 'd';
 89 		 *      t['e'] => 'f';
 90 		 */
 91 		mixin: function(obj, props) {
 92 
 93 			var host = this,
 94 				args = arguments,
 95 				i = 1,
 96 				o = obj || {},
 97 				len = args.length;
 98 
 99 			for (; i < len; i++) {
100 
101 				host._mixin(o, args[i]);
102 
103 			}
104 
105 			return o;
106 
107 		},
108 
109 		/**
110 		 * 合并多个对象为一个对象
111 		 * @param { Object } props 源对象.
112 		 * @return { Object } 合并后的对象.
113 		 * @spec Merge some object to one.
114 		 * @example
115 		 *      var o1 = { a: 'b' },
116 		 *      	o2 = { c: 'd' },
117 		 *      	o3 = { e: 'f' },
118 		 *      	o = object.merge(o1, o2, o3);
119 		 *      o['a'] => 'b';
120 		 *      o['c'] => 'd';
121 		 *      o['e'] => 'f';
122 		 */
123 		merge: function() {
124 
125 			var o = {};
126 
127 			return this.mixin.apply(this, [o].concat(lang.toArray(arguments)));
128 
129 		},
130 
131 		/**
132 		 * 转换对象为字符串
133 		 * @param { Object } obj 对象.
134 		 * @param { Array } symbol 分隔符, 数组中的第一个为 key, value 之间的间隔符, 第二个为各值之间的间隔.
135 		 * @param { Boolean } needEncode 是否需要编码中文.
136 		 * @return { String } 转换成的字符串.
137 		 * @spec Translate an object to string.
138 		 * @example
139 		 *      object.toString({ a: 'b', c: 'd' }) => 'a:b,c:d'
140 		 *      object.toString({ a: 'b', c: 'd' }, ['=', '&']) => 'a=b&c=d'
141 		 */
142 		toString: function(obj, symbol, needEncode) {
143 
144 			var i,
145 				arr = [],
146 				sym = symbol || [':', ','];
147 
148 			for (i in obj) {
149 
150 				arr.push(i + sym[0] + (!needEncode ? obj[i] : escape(obj[i].toString())));
151 
152 			}
153 
154 			return arr.join(sym[1]);
155 
156 		},
157 
158 		/**
159 		 * 转换字符串到对象
160 		 * @param { String } str 字符串.
161 		 * @param { Array } symbol 分隔符.
162 		 * @param { Boolean } needDecode 是否需要解码.
163 		 * @return { Object } 对象.
164 		 * @spec Translate a string to object.
165 		 * @example
166 		 *      var o1 = object.toObject('a:b,c:d'),
167 		 *      	o2 = object.toObject('a=b&c=d', ['=', '&']);
168 		 *      o1['a'] => 'b';
169 		 *      o1['c'] => 'd';
170 		 *      o2['a'] => 'b';
171 		 *      o2['c'] => 'd';
172 		 */
173 		toObject: function(str, symbol, needDecode) {
174 
175 			var	obj = {},
176 				sym = symbol || [':', ','],
177 				arr = str.split(sym[1]),
178 				i = 0,
179 				len = arr.length;
180 
181 			for (; i < len; i++) {
182 
183 				var kv = arr[i].split(sym[0]);
184 				obj[kv[0]] = !needDecode ? kv[1] : unescape(kv[1]);
185 
186 			}
187 
188 			return obj;
189 
190 		},
191 
192 		/**
193 		 * 获取对象
194 		 * @private
195 		 * @param { String } parts 名字.
196 		 * @param { Boolean } create 是否创建对象.
197 		 * @param { Object } context 上下文对象.
198 		 * @return { Object } 对象.
199 		 */
200 		_getProp: function(parts, create, context) {
201 
202 			var obj = context || win,
203 				len = parts.length,
204 				i = 0;
205 
206 			for (; i < len; i++) {
207 
208 				if (obj[parts[i]]) {
209 
210 					obj = obj[parts[i]];
211 					continue;
212 
213 				} else {
214 
215 					obj = create ? (obj[parts[i]] = {}) : 'undefined';
216 
217 				}
218 
219 			}
220 
221 			return obj;
222 
223 		},
224 
225 		/**
226 		 * 设置对象
227 		 * @param { String } name 名字.
228 		 * @param { Anything } value 值.
229 		 * @param { Object } context 上下文对象.
230 		 * @return { Object } 要设置的对象.
231 		 */
232 		set: function(name, value, context) {
233 
234 			var host = this,
235 				parts = name.split('.'),
236 				p = parts.pop(),
237 				obj = host._getProp(parts, true, context);
238 
239 			if (obj && p) {
240 
241 				return (obj[p] = value);
242 
243 			}
244 
245 		},
246 
247 		/**
248 		 * 根据字符串获取对象
249 		 * @param { String } name 名字.
250 		 * @param { Boolean } create 如果对象不存在是否创建.
251 		 * @param { Object } context 上下文对象.
252 		 * @return { Object } 对象.
253 		 * @spec set and get object.
254 		 * @example
255 		 *      object.set('nature.dom', { name: 'dom' });
256 		 *      var o = object.get('nature.dom');
257 		 *      o.name => 'dom';
258 		 */
259 		get: function(name, create, context) {
260 
261 			return this._getProp(name.split('.'), create, context);
262 
263 		}
264 
265 	};
