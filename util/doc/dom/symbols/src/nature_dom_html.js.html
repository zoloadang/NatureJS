 28 		 * 获取节点
 29 		 * @param { String | HTMLelement } id 可以是节点 id, 也可以是 DOM 节点.
 30 		 * @return { HTMLelement } 返回第一个节点或者 null.
 31 		 */
 32 		byId: function(id) {
 33 
 34 			if (!lang.isString(id)) {
 35 
 36 				return id;
 37 
 38 			}
 39 
 40 			if (browser.isIE) {
 41 
 42 				var host = this,
 43 					el = doc.getElementById(id);
 44 
 45 				if (el && (id === el.attributes.id.value || id === el.id)) {
 46 
 47 					//>=ie 8
 48 					return el;
 49 
 50 				} else {
 51 
 52 					//<=ie7
 53 					var eles = doc.all[id],
 54 						i = 0;
 55 
 56 					if (!eles || eles.nodeName) {
 57 
 58 						eles = [eles];
 59 
 60 					}
 61 
 62 					while (el = eles[i++]) {
 63 
 64 						if (id === el.attributes.id.value || id === el.id) {
 65 
 66 							return el;
 67 
 68 						}
 69 
 70 					}
 71 
 72 				}
 73 
 74 				return null;
 75 
 76 			} else {
 77 
 78 				return doc.getElementById(id) || null;
 79 
 80 			}
 81

 88 		 * @return { Array } 节点数组.

 91 
 92 			var host = this,
 93 				children = host.byId(node).childNodes,
 94 				ret = [],
 95 				i = 0,
 96 				len = children.length,
 97 				child;
 98 
 99 			for (; i < len; i++) {
100 
101 				child = children[i];
102 
103 				if (1 === child.nodeType) {
104 
105 					if (filter) {
106 
107 						filter(child) && ret.push(child);
108 
109 					} else {
110 
111 						ret.push(child);
112 
113 					}
114 
115 				}
116 
117 			}
118 
119 			return ret.length > 0 ? ret : null;

176 		/**
177 		 * 获取相邻节点
178 		 * @private
179 		 * @param { HTMLelement | String } node 目标节点或者 id.
180 		 * @param { Function } filter 过滤条件, 可选.
181 		 * @param { String } position 节点相对位置.
182 		 * @return { HTMLelement } 相邻节点或者 null.
183 		 */
184 		_getNeighbour: function(node, filter, position) {
185 
186 			var host = this,
187 				ele = host.byId(node),
188 				type = position + 'Sibling',
189 				neighbour,
190 				ret;
191 
192 			while (neighbour = ele[type]) {
193 
194 				if (1 === neighbour.nodeType) {
195 
196 					if (filter) {
197 
198 						if (filter(neighbour)) {
199 
200 							ret = neighbour;
201 							break;
202 
203 						}
204 
205 					} else {
206 
207 						ret = neighbour;
208 						break;
209 
210 					}
211 
212 				}
213 
214 				ele = ele[type];
215 
216 			}
217 
218 			return ret || null;

222 		/**

224 		 * @param { HTMLelement | String } node 节点或者节点 id.
225 		 * @param { Function } filter 过滤条件, 可选.
226 		 * @return { HTMLelement } 父节点或者 null.

228 		parent: function(node, filter) {
229 
230 			var host = this,
231 				root = doc.documentElement,
232 				ele = host.byId(node),
233 				par;
234 
235 			if (root != node) {
236 
237 				if (!filter) {
238

240 
241 				} else {
242 
243 					while (par = ele.parentNode) {
244 
245 						if (filter(par)) {
246 
247 							break;
248 
249 						} else {
250 
251 
252 							if (root == par) {
253 
254 								par = null;
255 								break;
256 
257 							} else {
258 
259 								ele = ele.parentNode;
260 
261 							}
262 
263 						}
264 
265 					}
266 
267 					return par;
268 
269 				}

280 		/**
281 		 * 判断一个节点是否是另一个节点的父节点
282 		 * @param { HTMLelement | String } ancestor 父节点或者 id.

284 		 * @return { Boolean } 如果是则返回 true, 否则返回 false.

286 		contains: function(ancestor, child) {
287 
288 			var host = this,
289 				a = host.byId(ancestor),
290 				c = host.byId(child);
291 
292 			if (!a || !c) {
293 
294 				return false;
295 
296 			}
297 
298 			//ie, opera, safari
299 			if (a['contains']) {
300 
301 				return a['contains'](c);
302 
303 			}
304 
305 			//firefox
306 			if (a['compareDocumentPosition']) {
307 
308 				return a === c || !!(a['compareDocumentPosition'](c) & 16);
309 
310 			}

386 		/**
387 		 * 将一个节点按指定位置放入
388 		 * @param { HTMLelement | String } node 新节点或者节点 id.
389 		 * @param { HTMLelement | String } refNode 目标节点或者节点 id.
390 		 * @param { String | Number } position 插入的位置, 可以是字符串 (before, after, replace, only, first, last), 默认为 last, 也可以是数字, 表示子元素的位置.
391 		 * @return { HTMLelement } 新节点.
392 		 */
393 		place: function(node, refNode, position) {
394 
395 			var host = this,
396 				n = host.byId(node),
397 				r = host.byId(refNode),
398 				par,
399 				children;
400 
401 			if (!n || !r) {
402 
403 				return false;
404 
405 			}
406 
407 			par = r.parentNode;
408 			children = host.children(r);
409 
410 			if (lang.isNumber(position)) {
411 
412 				if (position > children.length - 1) {
413 
414 					r.appendChild(n);
415 
416 				} else {
417 
418 					host._insertAfter(n, children[position - 1]);
419 
420 				}
421 
422 
423 			} else if (lang.isString(position) || !position) {
424 
425 				switch (position) {
426 
427 					case 'after':
428 						host._insertAfter(n, r);
429 						break;
430 
431 					case 'before':
432 						host._insertBefore(n, r);
433 						break;
434 
435 					case 'replace':
436 						par.replaceChild(n, r);
437 						break;
438 
439 					case 'only':
440 						host.empty(r);
441 						//continue
442 
443 					case 'first':
444 						if (host.first(r)) {
445 							host._insertBefore(n, host.first(r));
446 						}
447 						//else continue
448 					default:
449 						r.appendChild(n);
450 
451 				}
452 
453 			}
454 
455 			return n;
456

459 		/**
460 		 * 创建一个节点
461 		 * @TODO script tag, style tag, iframe, etc.
462 		 * @param { String } tag 需要创建的标签.
463 		 * @param { Object } attrs 标签属性, 可选.
464 		 * @param { HTMLelement | String } refNode 需要插入的节点或者节点 id, 可选.
465 		 * @param { String | Number } pos 需要插入的位置, 可选.
466 		 * @return { HTMLelement } 新创建的节点.
467 		 * @example
468 		 * 		html.create('div', { id: 'test' }, document.body, first);
469 		 */
470 		create: function(tag, attrs, refNode, pos) {
471 
472 			var host = this,
473 				isHTML = /<.+>/.test(tag),
474 				node,
475 				children,
476 				temp;
477 
478 			if (isHTML) {
479 
480 				//通过 html 字符串创建节点
481 				temp = doc.createElement('div');
482 				temp.innerHTML = tag;
483 				node = doc.createDocumentFragment();
484 
485 				while (children = host.children(temp)) {
486 
487 					node.appendChild(children[0]);
488 
489 				}
490 
491 				pos = refNode;
492 				refNode = attrs;
493 
494 			} else {
495 
496 				//通过 tag 创建节点
497 				node = doc.createElement(tag);
498 
499 				if (attrs) {
500 
501 					host.setAttribute(node, attrs);
502 
503 				}
504 
505 			}
506 
507 			if (refNode) {
508 
509 				return host.place(node, refNode, pos);
510 
511 			}
512 
513 			return node;
514

543 		 * @param { HTMLelement | String } node 节点或者节点 id.
544 		 * @param { String | Object } name attribute 名字, 如果为对象, 则循环对象设置 key/value.
545 		 * @param { String } value 需要设置的 attribute 值.
546 		 * @return { String | Object } 返回设置的值.

549 
550 			var host = this,
551 				el = host.byId(node),
552 				i;
553 
554 			if (lang.isString(name)) {
555 
556 				if ('innerHTML' === name) {
557 
558 					node[name] = value;
559 
560 				}
561 
562 				//兼容 ie7, 6
563 				if ('class' === name && browser.isIE < 8) {
564 
565 					name = 'className';
566 
567 				}
568 
569 				el.setAttribute(name, value);
570 
571 				return value;
572 
573 			} else if (lang.isObject(name)) {
574 
575 				for (i in name) {
576 
577 					host.setAttribute(el, i, name[i]);
578 
579 				}
580 
581 				return name;
582 
583 			}
