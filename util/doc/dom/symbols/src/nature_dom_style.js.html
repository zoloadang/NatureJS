  9 		doc = document,

 13 	/**
 14 	 * 样式操作

 20 
 21 		/**
 22 		 * @lends style

 25 
 26 		/**
 27 		 * 判断节点是否有 class
 28 		 * @param { HTMLelement | String } node 节点或者节点 id.
 29 		 * @param { String } cname class 名字.
 30 		 * @return { Boolean } 如果有则返回 true, 否则返回 false.
 31 		 */
 32 		hasClass: function(node, cname) {
 33 
 34 			var	reg = new RegExp('(?:^|\\s+)' + cname + '(?:\\s+|$)');
 35 
 36 			return reg.test(html.getAttribute(node, 'class'));
 37 
 38 		},
 39 
 40 		/**
 41 		 * 判断节点或者父节点是否有 class
 42 		 * @param { HTMLelement | String } node 节点或者节点 id.
 43 		 * @param { String } cname class 名字.
 44 		 * @return { Boolean } 如果有或者父节点有则返回 true, 否则返回 false.
 45 		 */
 46 		hasAncestor: function(node, cname) {
 47 
 48 			var host = this,
 49 				ret = host.hasClass(node, cname);
 50 
 51 			if (!ret) {
 52 
 53 				//尝试获取含有 class 的父节点
 54 				ret = html.parent(node, function(el) {
 55 
 56 					      return host.hasClass(el, cname);
 57 
 58 					  });
 59 
 60 			}
 61 
 62 			return !!ret;
 63 
 64 		},
 65 
 66 		/**
 67 		 * 为节点添加 class
 68 		 * @param { HTMLelement | String } node 节点或者节点 id.
 69 		 * @param { String } cname 需要添加的 class.
 70 		 * @return { String } 添加的 class.
 71 		 */
 72 		addClass: function(node, cname) {
 73 
 74 			var host = this,
 75 				cls = html.getAttribute(node, 'class'),
 76 				ret;
 77 
 78 			if (!host.hasClass(node, cname)) {
 79 
 80 				ret = string.trim(cls + SPACE + cname);
 81 				html.setAttribute(node, 'class', ret);
 82 
 83 			}
 84 
 85 			return cname;
 86 
 87 		},
 88 
 89 		/**
 90 		 * 移除节点 class
 91 		 * @param { HTMLelement | String } node 节点或者节点 id.
 92 		 * @param { String } cname 需要移除的 class.
 93 		 * @return { String } 移除的 class.
 94 		 */
 95 		removeClass: function(node, cname) {
 96 
 97 			var host = this,
 98 				cls,
 99 				reg,
100 				ret;
101 
102 			if (host.hasClass(node, cname)) {
103 
104 				reg = new RegExp('(?:^|\\s+)' + cname + '(?:\\s+|$)');
105 				cls = html.getAttribute(node, 'class');
106 				ret = string.trim(cls.split(reg).join(SPACE));
107 				html.setAttribute(node, 'class', ret);
108 
109 			}
110 
111 			return cname;
112 
113 		},
114 
115 		/**
116 		 * 添加或者移除节点 class
117 		 * @param { HTMLelement | String } node 节点或者节点 id.
118 		 * @param { String } cname 需要移除的 class.
119 		 * @return { String } 添加/移除的 class.
120 		 */
121 		toggleClass: function(node, cname) {
122 
123 			var	host = this,
124 				el = html.byId(node),
125 				supportToggle = !!('classList' in el)/* && !!('toggle' in el['classList'])*/;
126 
127 			if (supportToggle) {
128 
129 				return el['classList'].toggle(cname);
130 
131 			}
132 
133 			if (host.hasClass(node, cname)) {
134 
135 				return host.removeClass(node, cname);
136 
137 			}
138 
139 			return host.addClass(node, cname);
140 
141 		},
142 
143 		/**
144 		 * 获取节点 style
145 		 * @param { String | HTMLelement } node 节点或者节点 id.
146 		 * @param { String } name 需要获取的 style 名.
147 		 * @return { String } style 值.
148 		 * @example
149 		 * 		dom.getStyle('style', 'opacity');
150 		 */
151 		getStyle: function(node, name) {
152 
153 			var host = this,
154 				el = html.byId(node),
155 				property,
156 				value,
157 				computed,
158 				so = 'opacity' === name,
159 				sf = 'float' === name;
160 
161 			if (win['getComputedStyle']) {
162 
163 				property = sf ? 'cssFloat' : name;
164 				value = el['style'][property];
165 
166 				if (!value) {
167 
168 					computed = el['ownerDocument']['defaultView']['getComputedStyle'](el, null);
169 
170 					if (computed) {
171 
172 						value = computed[property];
173 
174 					}
175 
176 				}
177 
178 			} else if (el['currentStyle']) {
179 
180 				if (so) {
181 
182 					return host._getOpacity(el);
183 
184 				}
185 
186 				property = sf ? 'styleFloat' : name;
187 
188 				value = el['currentStyle'][property];
189 
190 			}
191 
192 			return host._fixStyle(el, name, value);
193 
194 		},
195 
196 		/**
197 		 * 修改获取的 style 值, 保持各浏览器一致
198 		 * @private
199 		 * @param { HTMLelement } node 节点.
200 		 * @param { String } name style 名.
201 		 * @param { String } value style 值.
202 		 * @return { String } style 值.
203 		 */
204 		_fixStyle: function(node, name, value) {
205 
206 			var host = this,
207 				_pixelRegExp = /margin|padding|width|height|max|min|offset/;
208 
209 			//fix auto, only fix width and height
210 			if ('auto' === value && browser.isIE < 8) {
211 				//非 ie 以及 ie8 以上浏览器下恒为 0
212 
213 				if ('width' === name) {
214 
215 					return node.offsetWidth;
216 
217 				}
218 
219 				if ('height' === name) {
220 
221 					return node.offsetHeight;
222 
223 				}
224 
225 			}
226 
227 			//fix px
228 			if (_pixelRegExp.test(name)) {
229 
230 				return host._fixPixel(value);
231 
232 			}
233 
234 			//@TODO fix inherit
235 			/*
236 			if ('inherit' === value && browser.isIE) {
237 
238 				if (docElem != node) {
239 
240 					return host.getStyle(node.parentNode, name);
241 
242 				}
243 
244 			}
245 			*/
246 
247 			return value;
248 
249 		},
250 
251 		/**
252 		 * 修改获取的 style 类型为 px 值的 style
253 		 * @private
254 		 * @param { String } value style 值.
255 		 * @return { String } style 值.
256 		 */
257 		_fixPixel: function(value) {
258 
259 			return parseFloat(value) || 0;
260 
261 		},
262 
263 		/**
264 		 * 获取透明度
265 		 * @private
266 		 * @param { HTMLelement } node 节点.
267 		 * @return { Number } opacity 值.
268 		 */
269 		_getOpacity: function(node) {
270 
271 			try {
272 
273 				return node['filters'][ASTR]['opacity'] / 100;
274 
275 			} catch (e) {
276 
277 				try {
278 
279 					return node['filters']['alpha']['opacity'] / 100;
280 
281 				} catch (e) {
282 
283 					return 1;
284 
285 				}
286 
287 			}
288 
289 		},
290 
291 		/**
292 		 * 设置节点 style
293 		 * @param { HTMLelement | String } node 节点或者节点 id.
294 		 * @param { String } name 属性名.
295 		 * @param { String } value 属性值.
296 		 * @return { String } value 值.
297 		 * @example:
298 		 * 		dom.setStyle('style', 'float', 'left');
299 		 * 		dom.setStyle(doc.body, { opacity: '0.2' });
300 		 */
301 		setStyle: function(node, name, value) {
302 
303 			var	host = this,
304 				el = html.byId(node),
305 				so = 'opacity' === name,
306 				sf = 'float' === name,
307 				property,
308 				i;
309 
310 			if (lang.isObject(name)) {
311 
312 				for (i in name) {
313 
314 					arguments.callee.call(host, node, i, name[i]);
315 
316 				}
317 
318 				return false;
319 
320 			}
321 
322 			if (so) {
323 
324 				return host._setOpacity(el, value);
325 
326 			}
327 
328 			if (sf) {
329 
330 				property = win['getComputedStyle'] ? 'cssFloat' : 'styleFloat';
331 
332 			} else {
333 
334 				property = name;
335 
336 			}
337 
338 			return el['style'][property] = value;
339 
340 
341 		}
342 
343 	};
