  6 define(['../type/lang.js', '../type/string.js', './html.js'], function(lang, string, html) {
  7 
  8 	var win = window,
  9 		doc = document,
 10 		SPACE = ' ';
 11 
 12 	/**
 13 	 * 样式操作
 14 	 * @name style
 15 	 * @namespace
 16 	 */
 17 
 18 	return {
 19 
 20 		/**
 21 		 * @lends style
 22 		 * @static
 23 		 */
 24 
 25 		/**
 26 		 * 判断节点是否有 class
 27 		 * @param { HTMLelement | String } node 节点或者节点 id.
 28 		 * @param { String } cname class 名字.
 29 		 * @return { Boolean } 如果有则返回 true, 否则返回 false.
 30 		 * @TODO 同时检测子元素
 31 		 */
 32 		hasClass: function(node, cname) {
 33 
 34 			var	reg = new RegExp('(?:^|\\s+)' + cname + '(?:\\s+|$)');
 35 
 36 			return reg.test(html.getAttribute(node, 'class'));
 37 
 38 		},
 39 
 40 		/**
 41 		 * 为节点添加 class
 42 		 * @param { HTMLelement | String } node 节点或者节点 id.
 43 		 * @param { String } cname 需要添加的 class.
 44 		 * @return { String } 添加的 class.
 45 		 */
 46 		addClass: function(node, cname) {
 47 
 48 			var host = this,
 49 				cls = html.getAttribute(node, 'class'),
 50 				ret;
 51 
 52 			if (!host.hasClass(node, cname)) {
 53 
 54 				ret = string.trim(cls + SPACE + cname);
 55 				html.setAttribute(node, 'class', ret);
 56 
 57 			}
 58 
 59 			return cname;
 60 
 61 		},
 62 
 63 		/**
 64 		 * 移除节点 class
 65 		 * @param { HTMLelement | String } node 节点或者节点 id.
 66 		 * @param { String } cname 需要移除的 class.
 67 		 * @return { String } 移除的 class.
 68 		 */
 69 		removeClass: function(node, cname) {
 70 
 71 			var host = this,
 72 				cls,
 73 				reg,
 74 				ret;
 75 
 76 			if (host.hasClass(node, cname)) {
 77 
 78 				reg = new RegExp('(?:^|\\s+)' + cname + '(?:\\s+|$)');
 79 				cls = html.getAttribute(node, 'class');
 80 				ret = string.trim(cls.split(reg).join(SPACE));
 81 				html.setAttribute(node, 'class', ret);
 82 
 83 			}
 84 
 85 			return cname;
 86 
 87 		},
 88 
 89 		/**
 90 		 * 添加或者移除节点 class
 91 		 * @param { HTMLelement | String } node 节点或者节点 id.
 92 		 * @param { String } cname 需要移除的 class.
 93 		 * @return { String } 添加/移除的 class.
 94 		 */
 95 		toggleClass: function(node, cname) {
 96 
 97 			var host = this;
 98 
 99 			if (host.hasClass(node, cname)) {
100 
101 				return host.removeClass(node, cname);
102 
103 			}
104 
105 			return host.addClass(node, cname);
106 
107 		},
108 
109 		/**
110 		 * 获取节点 style
111 		 * @param { String | HTMLelement } node 节点或者节点 id.
112 		 * @param { String } style 需要获取的 style 名.
113 		 * @return { String } style 值.
114 		 * @example
115 		 * 		dom.getStyle('style', 'opacity');
116 		 */
117 		getStyle: function(node, style) {
118 
119 			var host = this,
120 				el = html.byId(node),
121 				property,
122 				value,
123 				computed,
124 				so = 'opacity' === style,
125 				sf = 'float' === style;
126 
127 			if (win['getComputedStyle']) {
128 
129 				property = sf ? 'cssFloat' : style;
130 				value = el['style'][property];
131 
132 				if (!value) {
133 
134 					computed = el['ownerDocument']['defaultView']['getComputedStyle'](el, null);
135 
136 					if (computed) {
137 
138 						value = computed[property];
139 
140 					}
141 
142 				}
143 
144 				return value;
145 
146 			} else if (el['currentStyle']) {
147 
148 				if (so) {
149 
150 					return host._getOpacity(el);
151 
152 				}
153 
154 				property = sf ? 'styleFloat' : style;
155 
156 				return el['currentStyle'][property];
157 
158 			}
159 
160 		},
161 
162 		/**
163 		 * 获取透明度
164 		 * @private
165 		 * @param { HTMLelement } node 节点.
166 		 * @return { Number } opacity 值.
167 		 */
168 		_getOpacity: function(node) {
169 
170 			try {
171 
172 				return node['filters'][ASTR]['opacity'] / 100;
173 
174 			} catch (e) {
175 
176 				try {
177 
178 					return node['filters']['alpha']['opacity'] / 100;
179 
180 				} catch (e) {
181 
182 					return 1;
183 
184 				}
185 
186 			}
187 
188 		},
189 
190 		/**
191 		 * 设置节点 style
192 		 * @param { HTMLelement | String } node 节点或者节点 id.
193 		 * @param { String } name 属性名.
194 		 * @param { String } value 属性值.
195 		 * @return { String } value 值.
196 		 * @example:
197 		 * 		dom.setStyle('style', 'float', 'left');
198 		 * 		dom.setStyle(doc.body, { opacity: '0.2' });
199 		 */
200 		setStyle: function(node, name, value) {
201 
202 			var	host = this,
203 				el = html.byId(node),
204 				so = 'opacity' === name,
205 				sf = 'float' === name,
206 				property,
207 				i;
208 
209 			if (lang.isObject(name)) {
210 
211 				for (i in name) {
212 
213 					arguments.callee.call(host, node, i, name[i]);
214 
215 				}
216 
217 				return false;
218 
219 			}
220 
221 			if (so) {
222 
223 				return host._setOpacity(el, value);
224 
225 			}
226 
227 			if (sf) {
228 
229 				property = win['getComputedStyle'] ? 'cssFloat' : 'styleFloat';
230 
231 			} else {
232 
233 				property = name;
234 
235 			}
236 
237 			return el['style'][property] = value;
238 
239 
240 		}
241 
242 	};
243
