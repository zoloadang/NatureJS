1 /**
  2  * @fileoverview screen 模块.
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 
  6 define(['../bom/browser.js', './html.js'], function(browser, html) {
  7 
  8 	var win = window,
  9 		doc = document,
 10 		body = doc.body,
 11 		docElem = doc.documentElement,
 12 		OFFSET_WIDTH = 'offsetWidth',
 13 		OFFSET_Height = 'offsetHeight',
 14 		INNER_WIDTH = 'innerWidth',
 15 		INNER_HEIGHT = 'innerHeight',
 16 		CLIENT_WIDTH = 'clientWidth',
 17 		CLIENT_HEIGHT = 'clientHeight',
 18 		SCROLL_TOP = 'scrollTop',
 19 		SCROLL_LEFT = 'scrollLeft',
 20 		LEFT = 'left',
 21 		RIGHT = 'right',
 22 		TOP = 'top',
 23 		BOTTOM = 'bottom';
 24 
 25 	/**
 26 	 * 文档尺寸, 节点位置获取
 27 	 * @name screen
 28 	 * @namespace
 29 	 */
 30 
 31 	return {
 32 
 33 		/**
 34 		 * @lends screen
 35 		 * @static
 36 		 */
 37 
 38 		/**
 39 		 * 获取滚动条已滚的范围
 40 		 * @return { Object } 返回已滚的范围对象.
 41 		 */
 42 		scroll: function() {
 43 
 44 			var node = browser.isQuirks ? body : docElem;
 45 
 46 			return { x: node[SCROLL_LEFT] || 0, y: node[SCROLL_TOP] || 0 };
 47 
 48 		},
 49 
 50 		/**
 51 		 * 获取纵向滚动条已滚范围
 52 		 * @return { Number } 返回纵向滚动条已滚范围.
 53 		 */
 54 		scrollTop: function() {
 55 
 56 			return this.scroll()['y'];
 57 
 58 		},
 59 
 60 		/**
 61 		 * 获取横向滚动条已滚范围
 62 		 * @return { Number } 返回横向滚动条已滚范围.
 63 		 */
 64 		scrollLeft: function() {
 65 
 66 			return this.scroll()['x'];
 67 
 68 		},
 69 
 70 		/**
 71 		 * 获取节点位置
 72 		 * @param { HTMLelement | String } node 节点或者节点 id.
 73 		 * @param { Boolean } includeScroll 计算时是否包含滚动条.
 74 		 * @return { Object } { x: 100, y: 100, w: 100, h: 100 }.
 75 		 */
 76 		position: function(node, includeScroll) {
 77 
 78 			var host = this,
 79 				scroll,
 80 				el = html.byId(node),
 81 				coord = el.getBoundingClientRect(),
 82 				ret = { x: coord[LEFT], y: coord[TOP], w: coord[RIGHT] - coord[LEFT], h: coord[BOTTOM] - coord[TOP] };
 83 
 84 			if (includeScroll) {
 85 
 86 				scroll = host.scroll();
 87 
 88 				ret['x'] += scroll['x'];
 89 				ret['y'] += scroll['y'];
 90 
 91 			}
 92 
 93 			return ret;
 94 
 95 		},
 96 
 97 		/**
 98 		 * 获取文档大小
 99 		 * @return { Array } 数组, 第一个是文档宽度, 第二个是文档高度.
100 		 */
101 		docSize: function() {
102 
103 			return [
104 
105 				Math.max(body[OFFSET_WIDTH], docElem[OFFSET_WIDTH]),
106 				Math.max(body[OFFSET_HEIGHT], docElem[OFFSET_HEIGHT])
107 
108 			];
109 
110 		},
111 
112 		/**
113 		 * 获取文档高度
114 		 * @return { Number } 文档高度.
115 		 */
116 		docHeight: function() {
117 
118 			return this.docSize()[1];
119 
120 		},
121 
122 		/**
123 		 * 获取文档宽度
124 		 * @return { Number } 文档宽度.
125 		 */
126 		docWidth: function() {
127 
128 			return this.docSize()[0];
129 
130 		},
131 
132 		/**
133 		 * 获取文档可视大小
134 		 * @return { Array } 文档可视大小, 第一个元素为可视宽度, 第二个为可视高度.
135 		 */
136 		viewport: function() {
137 
138 			return [
139 
140 				win[INNER_WIDTH] || docElem[CLIENT_WIDTH],
141 				win[INNER_HEIGHT] || docElem[CLIENT_HEIGHT]
142 
143 			];
144 
145 		},
146 
147 		/**
148 		 * 获取文档可视宽度
149 		 * @return { Number } 文档可视宽度.
150 		 */
151 		viewportWidth: function() {
152 
153 			return this.viewport()[0];
154 
155 		},
156 
157 		/**
158 		 * 获取文档可视高度
159 		 * @return { Number } 文档可视高度.
160 		 */
161 		viewportHeight: function() {
162 
163 			return this.viewport()[1];
164 
165 		}
166 
167 	};
168 
169 });
