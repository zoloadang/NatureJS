1 /**
  2  * @fileoverview 类的声明, 继承, 模拟多继承(聚合).
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 define(['../type/lang.js', '../type/array.js', '../type/object', './extend.js'], function(lang, array, object, extend) {
  6 
  7 	/**
  8 	 * 类的创建, 继承.
  9 	 * @name class
 10 	 * @namespace
 11 	 */
 12 
 13 	var count = 0,
 14 		prefix = 'Nature_Class_',
 15 		DECLARED_CLASS = 'declaredClass';
 16 
 17 	/**
 18 	 * @lends class
 19 	 * @static
 20 	 */
 21 
 22 	/**
 23 	 * 获取 mro 序列.
 24 	 * @private
 25 	 * @param { Array } cls 超类集合.
 26 	 * @return { Array } base 集合.
 27 	 * @see http://www.python.org/download/releases/2.3/mro/
 28 	 */
 29 	function c3mro(bases) {
 30 
 31 		var i, len, base,
 32 			_meta, lin, j, proto, rec, name, top,
 33 			namemap = {}, clsCount = 1,
 34 			result = [], refs,
 35 			roots = [{cls: 0, refs: []}];//与 namemap 对应
 36 
 37 		// build a list of bases naming them if needed
 38 		for (i = 0, len = bases.length; i < len; i++) {
 39 
 40 			base = bases[i];
 41 			_meta = base['_meta'];
 42 			lin = _meta ? _meta['bases'] : [base];
 43 
 44 			top = 0;
 45 
 46 			for (j = lin.length - 1; j >= 0; j--) {
 47 
 48 				proto = lin[j].prototype;
 49 
 50 				if (!proto.hasOwnProperty(DECLARED_CLASS)) {
 51 					proto[DECLARED_CLASS] = prefix + count++;
 52 				}
 53 
 54 				name = proto[DECLARED_CLASS];
 55 
 56 				if (!namemap.hasOwnProperty(name)) {
 57 
 58 					namemap[name] = { count: 0, refs: [], cls: lin[j]};
 59 					clsCount++;
 60 
 61 				}
 62 
 63 				rec = namemap[name];
 64 
 65 				if (top && top !== rec) {
 66 					rec.refs.push(top);
 67 					++top.count;
 68 				}
 69 
 70 				top = rec;
 71 
 72 			}
 73 
 74 			++top.count;
 75 			roots[0].refs.push(top);
 76 
 77 		}
 78 
 79 		// remove classes without external references recursively
 80 		while (roots.length) {
 81 			top = roots.pop();
 82 			result.push(top.cls);
 83 			--clsCount;
 84 
 85 			while (refs = top.refs, refs.length == 1) {
 86 				top = refs[0];
 87 				//end
 88 				if (!top || --top.count) {
 89 					top = 0;
 90 					break;
 91 				}
 92 				result.push(top.cls);
 93 				--clsCount;
 94 			}
 95 
 96 			if (top) {
 97 				for (i = 0, len = refs.length; i < len; i++) {
 98 					top = refs[i];
 99 					if (!--top.count) {
100 						roots.push(top);
101 					}
102 				}
103 			}
104 
105 		}
106 
107 		if (clsCount) {
108 			throw new Error('error');
109 		}
110 
111 		// calculate the superclass offset
112 		base = bases[0];
113 		result[0] = base ?
114 					base._meta && base === result[result.length - base._meta.bases.length] ? base._meta.bases.length : 1
115 					: 0;
116 
117 		return result;
118 
119 	}
120 
121 	/**
122 	 * 检测是否是子类.
123 	 * @private
124 	 * @param { Function } superclass 超类.
125 	 * @return 如果是, 返回 true, 否则 返回 false.
126 	 */
127 	function isInstanceOf(superclass) {
128 
129 		var host = this;
130 		return array.indexOf(host.constructor['_meta']['bases'], superclass) > -1 || host instanceof superclass;
131 
132 	}
133 
134 	/**
135 	 * 创建一个构造函数, 并从超类或者一个聚合继承属性, 方法.
136 	 * @param { String } name 类名.
137 	 * @param { Array | Function } superclass 多个超类或者一个超类, 如果是数组, 那么默认第一个为超类.
138 	 * @param { Object } props 类包含的方法.
139 	 * @return { Object } 新生成的 constructor.
140 	 * @spec Inherited from one Class.
141 	 * @example
142 	 * 		var A = function() {};
143 	 * 		var B = declare(A);
144 	 * 		var c = new B;
145 	 * 		c instanceof A => true
146 	 * 		c.isInstanceOf(A) => true
147 	 *
148 	 * @spec Inherited from multi Classes.
149 	 * @example
150 	 * 		var A = declare(null);
151 	 * 		var B = declare(A);
152 	 * 		var C = declare([B, A]);
153 	 * 		var D = declare('haha', [C, B, A]);
154 	 * 		var c = new C;
155 	 * 		var d = new D;
156 	 * 		c instanceof A => true
157 	 * 		c instanceof B => false
158 	 * 		c instanceof C => true
159 	 * 		c.isInstanceOf(A) => true
160 	 * 		c.isInstanceOf(B) => true
161 	 * 		c.isInstanceOf(C) => true
162 	 * 		d instanceof A => true
163 	 * 		d instanceof B => true
164 	 * 		d instanceof C => true
165 	 * 		d instanceof D => true
166 	 * 		d.isInstanceOf(A) => true
167 	 * 		d.isInstanceOf(B) => true
168 	 * 		d.isInstanceOf(C) => true
169 	 * 		d.isInstanceOf(D) => true
170 	 * 		d.declaredClass => 'haha'
171 	 */
172 	function declare(name, superclass, props) {
173 
174 		var proto = {},
175 			ctor = new Function,
176 			mixin = 1,
177 			bases, i, len, obj = {};
178 
179 		if (!lang.isString(name)) {
180 
181 			props = superclass;
182 			superclass = name;
183 
184 		}
185 
186 		if (lang.isArray(superclass)) {
187 
188 			bases = c3mro(superclass);
189 
190 			mixin = bases.length - bases[0];
191 
192 			superclass = bases[mixin];
193 
194 		} else {
195 
196 			bases = [0];
197 			if (superclass) {
198 				bases.push(superclass);
199 			}
200 
201 		}
202 
203 		if (superclass) {
204 
205 			for (i = mixin - 1; i > 0; i--) {
206 
207 				object.mixin(obj, bases[i].prototype);
208 
209 			}
210 
211 			props = object.mixin(obj, props);
212 
213 		}
214 
215 		object.mixin(proto, props);
216 
217 		//add standard method to the prototype, must be added before extend
218 		proto.isInstanceOf = isInstanceOf;
219 
220 		//extend
221 		extend(ctor, superclass, proto, { extend: extend });
222 
223 		bases[0] = ctor;
224 		ctor['_meta'] = { bases: bases, parent: superclass };
225 
226 		proto = ctor.prototype;
227 		//add declaredClass
228 		if (name && lang.isString(name)) {
229 			object.set(name, ctor);
230 			proto[DECLARED_CLASS] = name;
231 		}
232 
233 		return ctor;
234 
235 	};
236 
237 	return declare;
238 
239 });
