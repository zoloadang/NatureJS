  2  * @fileoverview 类的声明, 继承.

  5 define(['../type/lang.js', '../type/array.js', '../type/object', './extend.js'], function(lang, array, object, extend) {
  6 
  7 	/**
  8 	 * 创建一个构造函数
  9 	 * @name class
 10 	 * @namespace
 11 	 */
 12 
 13 	/**
 14 	 * @lends class 
 15 	 * @static
 16 	 */
 17 
 18 	/**
 19 	 * 获取 base
 20 	 * @private
 21 	 * @param { Array } cls 超类集合.
 22 	 * @return { Array } base 集合.
 23 	 */
 24 	function _getBases(cls) {
 25 
 26 		var i = 0,
 27 			len = cls.length,
 28 			ret = [], base;
 29 
 30 		for (; i < len; i++) {
 31 
 32 			base = cls[i];
 33 
 34 			var _meta = base['_meta'],
 35 				cli = [];
 36 
 37 			if (_meta) {
 38 
 39 				cli = _meta['bases'];
 40 
 41 			} else {
 42 
 43 				0 !== i && (cli = [base]);
 44 
 45 			}
 46 
 47 			if (cli && cli.length) {
 48 
 49 				ret = ret.concat(cli);
 50 
 51 			}
 52 
 53 		}
 54 
 55 		return ret;
 56 
 57 	}
 58 
 59 	/**
 60 	 * 检测是否是子类
 61 	 * @private
 62 	 * @param { Function } superclass 超类.
 63 	 * @return 如果是, 返回 true, 否则 返回 false.
 64 	 */
 65 	function isInstanceOf(superclass) {
 66 
 67 		var host = this;
 68 
 69 		return array.indexOf(host.constructor['_meta']['bases'], superclass) > -1 || host instanceof superclass;
 70 
 71 	}
 72 
 73 	/**
 74 	 * 创建一个构造函数
 75 	 * @param { String } name 类名.
 76 	 * @param { Array | Function } superclass 多个超类或者一个超类, 如果是数组, 那么默认第一个为超类.
 77 	 * @param { Object } props 类包含的方法.
 78 	 * @return { Object } 新生成的 constructor.
 79 	 * @spec Inherited from one Class.
 80 	 * @example
 81 	 * 	var A = function() {};
 82 	 * 	var B = declare(A);
 83 	 * 	var c = new B;
 84 	 * 	c instanceof A => true
 85 	 * 	c.isInstanceOf(A) => true
 86 	 *
 87 	 * @spec Inherited from multi Classes.
 88 	 * 	var A = declare(null);
 89 	 * 	var B = declare(A);
 90 	 * 	var C = declare([A, B]);
 91 	 * 	var d = new C;
 92 	 * 	d instanceof A => true
 93 	 * 	d instanceof B => false
 94 	 * 	d instanceof C => true
 95 	 * 	d.isInstanceOf(A) => true
 96 	 * 	d.isInstanceOf(B) => true
 97 	 * 	d.isInstanceOf(C) => true
 98 	 */
 99 	function declare(name, superclass, props) {
100 
101 		var proto = {},
102 			ctor,
103 			cls, i, len, obj = {};
104 
105 		if (!lang.isString(name)) {
106 
107 			props = superclass;
108 			superclass = name;
109 			ctor = function() {};
110 
111 		} else {
112 
113 			ctor = object.get(name, true);
114 
115 		}
116 
117 		//add meta
118 		ctor['_meta'] = { bases: [] };
119 
120 		if (lang.isArray(superclass)) {
121 
122 			cls = _getBases(superclass);
123 
124 			superclass = superclass[0];
125 
126 			for (i = 0, len = cls.length; i < len; i++) {
127 
128 				object.mixin(obj, cls[i].prototype);
129 				ctor['_meta']['bases'].push(cls[i]);
130 
131 			}
132 
133 			props = object.mixin(obj, props);
134 
135 		}
136 
137 		object.mixin(proto, props);
138 
139 		//add standard method to the prototype, must be added before extend
140 		proto.isInstanceOf = isInstanceOf;
141 
142 		//extend
143 		extend(ctor, superclass, proto, { extend: extend });
144 
145 		return ctor;
146 
147 	};
148 
149 	return declare;
150 
151 });
