  5 define(['../type/object.js', '../type/string.js'], function(object, string) {
  6 
  7 	var win = window,
  8 		doc = document;
  9 
 10 	/**
 11 	 * cookie 模块
 12 	 * @name cookie
 13 	 * @namespace
 14 	 */
 15 	return {
 16 
 17 		/**
 18 		 * @lends cookie
 19 		 * @static
 20 		 */
 21 
 22 		/**
 23 		 * 获取 cookie
 24 		 * @param { String } name cookie 名.
 25 		 * @return { String } cookie 值.
 26 		 */
 27 		get: function(name) {
 28 
 29 			var cookie = doc.cookie,
 30 				reg = new RegExp('(?:^|;\\s*)' + unescape(string.trim(name)) + '=([^;]*)'),
 31 				match = cookie.match(reg);
 32 
 33 			return match ? decodeURIComponent(match[1]) : undefined;
 34 
 35 	    },
 36 
 37 		/**
 38 		 * 设置 cookie
 39 		 * @param { String } name cookie 名.
 40 		 * @param { String } value cookie 值.
 41 		 * @param { Object } props 其它选项.
 42 		 * @return { String } cookie 值.
 43 		 */
 44 		set: function(name, value, props) {
 45 
 46 			props = props || {};
 47 
 48 			var str = escape(name) + '=' + encodeURIComponent(value) + ';',
 49 				exp = props['expires'],
 50 				age = props['max-age'];
 51 
 52 			if (exp) {
 53 
 54 				props['expires'] = exp.toUTCString();
 55 
 56 			}
 57 
 58 			if (age) {
 59 
 60 				props['max-age'] = age * 24 * 60 * 60;
 61 
 62 			}
 63 
 64 			doc.cookie = str + object.toString(props, ['=', ';']);
 65 
 66 		},
 67 
 68 		/**
 69 		 * 检测浏览器是否支持 cookie
 70 		 * @return { Boolean } 如果浏览器开启了 cookie. 则返回 true, 否则返回 false.
 71 		 */
 72 		isEnable: function() {
 73 
 74 			var host = this,
 75 				nav = win.navigator,
 76 				name = '_natureCookieTest',
 77 				value = '_natureCookieAllowed';
 78 
 79 			if (!('cookieEnabled' in nav)) {
 80 
 81 				//尝试设置 cookie
 82 				host.set(name, value);
 83 				nav['cookieEnabled'] = value === host.get(name);
 84 
 85 				//删除 cookie
 86 				if (nav['cookieEnabled']) {
 87 
 88 					host.set(name, '', { 'max-age': -1 });
 89 
 90 				}
 91 
 92 			}
 93 
 94 			return nav['cookieEnabled'];
 95 
 96 		}
 97 
 98 	};
 99 
100 });
