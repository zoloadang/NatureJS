1 /**
  2  * @fileoverview 浏览器信息.
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 
  6 define(['../type/lang.js'], function(lang) {
  7 
  8 	/**
  9 	 * 浏览器信息
 10 	 * @name browser
 11 	 * @namespace
 12 	 * @require lang.js
 13 	 */
 14 	var win = window,
 15 		doc = document,
 16 		n = navigator,
 17 		ua = n.userAgent,
 18 		av = n.appVersion,
 19 		tv = parseFloat(av),
 20 		UNDEFINED = undefined,
 21 		opera = win.opera,
 22 		mode,
 23 		browser = {};
 24 
 25 	/**
 26 	 * @lends browser
 27 	 * @static
 28 	 */
 29 
 30 	if (!lang.isUndefined(win)) {
 31 
 32 		/**
 33 		 * 是否是浏览器
 34 		 * @property
 35 		 */
 36 		browser.isBrowser = true;
 37 
 38 		/**
 39 		 * 是否是 Opera
 40 		 * @property
 41 		 */
 42 		browser.isOpera = opera ? parseFloat(opera.version()) : UNDEFINED;
 43 
 44 		/**
 45 		 * 是否是 Khtml 浏览器
 46 		 * @proprety
 47 		 */
 48 		browser.isKhtml = av.indexOf('Konqueror') > -1 ? tv : UNDEFINED;
 49 
 50 		/**
 51 		 * 是否是 WebKit 浏览器
 52 		 * @property
 53 		 */
 54 		browser.isWebKit = parseFloat(ua.split('WebKit/')[1]) || UNDEFINED;
 55 
 56 		/**
 57 		 * 是否是 Chrome 浏览器
 58 		 * @property
 59 		 */
 60 		browser.isChrome = parseFloat(ua.split('Chrome/')[1]) || UNDEFINED;
 61 
 62 		/**
 63 		 * 是否是 Maxthon 浏览器
 64 		 * @property
 65 		 */
 66 		browser.isMaxthon = parseFloat(ua.split('Maxthon/')[1]) || UNDEFINED;
 67 
 68 		/**
 69 		 * 是否是 Safari 浏览器
 70 		 * @property
 71 		 */
 72 		if (browser.isWebKit && !browser.isChrome && !browser.isMaxthon && ua.indexOf('Safari') > -1) {
 73 
 74 			browser.isSafari = parseFloat(ua.split('Version/')[1]) || UNDEFINED;
 75 
 76 		}
 77 
 78 		/**
 79 		 * 是否是 Gecko 核心浏览器 (等同于 isMozilla, isMoz)
 80 		 * @property
 81 		 */
 82 		browser.isGecko = browser.isMozilla = browser.isMoz = ua.indexOf('Gecko') > -1 && !browser.isWebKit && !browser.isKhtml ? tv : UNDEFINED;
 83 
 84 		/**
 85 		 * 是否是 Firefox 浏览器 (等同于 isFF)
 86 		 * @property
 87 		 */
 88 		browser.isFirefox = browser.isFF = browser.isGecko ? parseFloat(ua.split('Firefox/')[1]) : UNDEFINED;
 89 
 90 		/**
 91 		 * 是否是 IE 浏览器
 92 		 * @property
 93 		 */
 94 		if (doc.all && !browser.isOpera) {
 95 
 96 			mode = doc.docMode;
 97 
 98 			browser.isIE = mode && 5 !== mode ? mode : parseFloat(ua.split('MSIE ')[1]);
 99 
100 		}
101 
102 		/**
103 		 * 是否是怪异模式
104 		 * @property
105 		 */
106 		browser.isQuirks = 'BackCompat' === doc.compatMode;
107 
108 	}
109 
110 	return browser;
111 
112 });
