1 /**
  2  * @fileoverview uri 处理.
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 define(['../type/lang.js', '../type/object.js'], function(lang, object) {
  6 
  7 	var	h = location.host,
  8 		QMARK = '?',
  9 		AND = '&';
 10 
 11 	/**
 12 	 * @name uri
 13 	 * @namespace
 14 	 */
 15 	return {
 16 
 17 		/**
 18 		 * @lends uri
 19 		 * @static
 20 		 */
 21 
 22 		/**
 23 		 * 检查是否包含 ?
 24 		 * @param { String } url 需要检查的 url.
 25 		 * @return { Boolean } 如果含有 ?, 返回 true, 否则返回 false.
 26 		 */
 27 		checkQmark: function(url) {
 28 
 29 			return url.indexOf(QMARK) > -1;
 30 
 31 		},
 32 
 33 		/**
 34 		 * 添加时间戳
 35 		 * @param { String } url 需要添加时间戳的 url.
 36 		 * @return { String } 添加时间戳后的 url.
 37 		 * @example:
 38 		 * 		uri.addStamp('http://www.12sui.cn/'); //print: http://www.12sui.cn/?t=1232421394
 39 		 */
 40 		addStamp: function(url) {
 41 
 42 			return url + (this.checkQmark(url) ? AND : QMARK) + 't=' + new Date().getTime();
 43 
 44 		},
 45 
 46 		/**
 47 		 * 构建 url
 48 		 * @param { String } url 初始 url.
 49 		 * @param { String | Object } params 要拼接的参数串.
 50 		 * @return { String } 构建好的字符串.
 51 		 * @example
 52 		 * 		uri.buildUri('http://www.12sui.cn/', { page: '1' }) // http://www.12sui.cn/?page=1
 53 		 * 		uri.buildUri('http://www.12sui.cn/', 'page=1') // http://www.12sui.cn/?page=1
 54 		 */
 55 		buildUri: function(url, params) {
 56 
 57 			var mark = this.checkQmark(url) ? AND : QMARK;
 58 
 59 			return url + mark + (lang.isObject(params) ? object.param(params) : params);
 60 
 61 		}
 62 
 63 	};
 64 
 65 });
