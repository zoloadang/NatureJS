1 /**
  2  * @fileoverview uri 处理.
  3  * @author nanzhi<nanzhienai@163.com>
  4  */
  5 define(['../type/lang.js', '../type/object.js'], function(lang, object) {
  6 
  7 	var	h = location.host,
  8 		QMARK = '?',
  9 		AND = '&';
 10 
 11 	/**
 12 	 * @name uri
 13 	 * @namespace
 14 	 */
 15 	return {
 16 
 17 		/**
 18 		 * @lends uri
 19 		 * @static
 20 		 */
 21 
 22 		/**
 23 		 * 检查是否包含 ?
 24 		 * @param { String } url 需要检查的 url.
 25 		 * @return { Boolean } 如果含有 ?, 返回 true, 否则返回 false.
 26 		 * @spec check is uri has ?.
 27 		 * @example
 28 		 *      uri.checkQmark('http://www.12sui.cn/') => false
 29 		 *      uri.checkQmark('http://www.12sui.cn/?page=1') => true
 30 		 */
 31 		checkQmark: function(url) {
 32 
 33 			return url.indexOf(QMARK) > -1;
 34 
 35 		},
 36 
 37 		/**
 38 		 * 添加时间戳
 39 		 * @param { String } url 需要添加时间戳的 url.
 40 		 * @return { String } 添加时间戳后的 url.
 41 		 * @spec add time stamp to an url.
 42 		 * @example
 43 		 *      uri.addStamp('http://www.12sui.cn/').indexOf('http://www.12sui.cn/?t=') > -1 => true
 44 		 */
 45 		addStamp: function(url) {
 46 
 47 			return url + (this.checkQmark(url) ? AND : QMARK) + 't=' + new Date().getTime();
 48 
 49 		},
 50 
 51 		/**
 52 		 * 构建 url
 53 		 * @param { String } url 初始 url.
 54 		 * @param { String | Object } params 要拼接的参数串.
 55 		 * @return { String } 构建好的字符串.
 56 		 * @spec build a url by some params.
 57 		 * @example
 58 		 *      uri.buildUri('http://www.12sui.cn/', { page: '1' }) => 'http://www.12sui.cn/?page=1'
 59 		 *      uri.buildUri('http://www.12sui.cn/', 'page=1') => 'http://www.12sui.cn/?page=1'
 60 		 */
 61 		buildUri: function(url, params) {
 62 
 63 			var host = this,
 64 				mark = host.checkQmark(url) ? AND : QMARK;
 65 
 66 			return url + mark + (lang.isObject(params) ? host.param(params) : params);
 67 
 68 		},
 69 
 70 		/**
 71 		 * 转换对象为 url 参数
 72 		 * @param { Object } obj 对象.
 73 		 * @return { String } url 参数字符串.
 74 		 * @spec Translate object to param string.
 75 		 * @example
 76 		 *      uri.param({a:'b', c:'d'}) => 'a=b&c=d';
 77 		 */
 78 		param: function(obj) {
 79 
 80 			return object.toString(obj, ['=', '&'], true);
 81 
 82 		},
 83 
 84 		/**
 85 		 * 转换 url 参数为字符串
 86 		 * @param { String } str url 参数字符串.
 87 		 * @return { Object } 对象.
 88 		 * @spec Translate param string to object.
 89 		 * @example
 90 		 *      var o = uri.unparam('a=b&c=d');
 91 		 *      o['a'] => 'b';
 92 		 *      o['c'] => 'd';
 93 		 */
 94 		unparam: function(str) {
 95 
 96 			return object.toObject(str, ['=', '&'], true);
 97 
 98 		}
 99 
100 	};
101 
102 });
